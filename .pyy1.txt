def main():
    # --- Input Section ---
    name = input("Enter your name: ")
    age = int(input("Enter your age: "))

    # --- Processing Section ---
    future_age = age + 10

    # --- Output Section ---
    print(f"\nHello {name}! After 10 years, you will be {future_age} years old.")

# --- Main Function Call ---
if __name__ == "__main__":
    main()
